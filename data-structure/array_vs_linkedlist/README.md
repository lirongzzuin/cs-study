    constructor(value) {
        this.value = value;
        this.next = null;
    }

class LinkedList {
    constructor() {
        this.head = null;
    }
    add(value) {
        const newNode = new ListNode(value);
        if (!this.head) this.head = newNode;
        else {

    }
